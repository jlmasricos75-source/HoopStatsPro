<script>

const STATE = {

  game: {

    team1: {name: 'Mi Equipo', score: 0},

    team2: {name: 'Rival', score: 0},

    period: 1,

    time: 600,

    started: false

  },

  players: [

    {id:1, num:4, name:'García', onCourt:false, pts:0, reb:0, ast:0, stl:0},

    {id:2, num:7, name:'López', onCourt:false, pts:0, reb:0, ast:0, stl:0},

    {id:3, num:9, name:'Martín', onCourt:false, pts:0, reb:0, ast:0, stl:0},

    {id:4, num:12, name:'Pérez', onCourt:false, pts:0, reb:0, ast:0, stl:0},

    {id:5, num:15, name:'Sánchez', onCourt:false, pts:0, reb:0, ast:0, stl:0}

  ],

  history: JSON.parse(localStorage.getItem('hoopstat_history')) || [],

  videoTime: 0

};


// Inicializar

document.addEventListener('DOMContentLoaded', function() {

  setupNav();

  loadHistory();

  updateDisplay();

});


function setupNav() {

  document.querySelectorAll('.nav-btn').forEach(btn => {

    btn.addEventListener('touchstart', function(e) {

      showTab(this.dataset.tab);

      e.preventDefault();

    }, {passive: false});

    btn.addEventListener('click', function(e) {

      showTab(this.dataset.tab);

    });

  });

}


function showTab(tab) {

  document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));

  document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));

  document.getElementById(tab).classList.add('active');

  document.querySelector(`[data-tab="${tab}"]`).classList.add('active');

}


function startGame() {

  const team1 = document.getElementById('my-team').value || 'Mi Equipo';

  const team2 = document.getElementById('rival-team').value || 'Rival';

  

  STATE.game.team1.name = team1;

  STATE.game.team2.name = team2;

  STATE.game.started = true;

  

  showTab('live');

  updateDisplay();

}


function updateDisplay() {

  // Marcador

  document.getElementById('team1-score').textContent = STATE.game.team1.score;

  document.getElementById('team2-score').textContent = STATE.game.team2.score;

  document.getElementById('team1-name').textContent = STATE.game.team1.name;

  document.getElementById('team2-name').textContent = STATE.game.team2.name;

  

  const mins = Math.floor(STATE.game.time / 60);

  const secs = (STATE.game.time % 60).toString().padStart(2, '0');

  document.getElementById('period-info').textContent = `${STATE.game.period}er Cuarto · ${mins}:${secs}`;


  // Jugadores

  renderPlayers();

  renderCourt();

}


function renderPlayers() {

  let html = '';

  STATE.players.forEach(p => {

    const stats = `${p.pts}pt ${p.reb}r ${p.ast}a`;

    html += `

      <div class="player-card ${p.onCourt ? 'active' : ''}">

        <div class="player-num">${p.num}</div>

        <div class="player-name">${p.name}</div>

        <div class="stats-line">

          <span class="stat-badge">${stats}</span>

        </div>

        <button class="btn btn-primary" style="width:100%; margin-top:8px; font-size:14px;"

          ontouchstart="togglePlayer(${p.id}); return false;" 

          onclick="togglePlayer(${p.id})">

          ${p.onCourt ? 'FUERA' : 'PISTA'}

        </button>

      </div>

    `;

  });

  document.getElementById('players-grid').innerHTML = html;

}


function renderCourt() {

  let html = '';

  STATE.players.filter(p => p.onCourt).forEach(p => {

    html += `<div class="court-player" ontouchstart="selectPlayer(${p.id});return false;" onclick="selectPlayer(${p.id})">${p.num}</div>`;

  });

  document.getElementById('court-players').innerHTML = html;

  

  // También en vídeo

  document.getElementById('video-court').innerHTML = html;

}


function togglePlayer(id) {

  const player = STATE.players.find(p => p.id === id);

  player.onCourt = !player.onCourt;

  updateDisplay();

}


function selectPlayer(id) {

  // Acción para jugador seleccionado en pista

  console.log('Jugador seleccionado:', id);

}


function addPoints(points, team = 1) {

  if (team === 1) {

    STATE.game.team1.score += points;

    // Asignar a jugador activo

    const activePlayer = STATE.players.find(p => p.onCourt);

    if (activePlayer) activePlayer.pts += points;

  } else {

    STATE.game.team2.score += points;

  }

  updateDisplay();

}


function periodNext() {

  STATE.game.period++;

  STATE.game.time = 600;

  if (STATE.game.period > 4) {

    saveGame();

    alert('¡Partido terminado!');

    resetGame();

  }

  updateDisplay();

}


function resetGame() {

  STATE.game.team1.score = 0;

  STATE.game.team2.score = 0;

  STATE.game.period = 1;

  STATE.game.time = 600;

  STATE.players.forEach(p => {

    p.pts = 0; p.reb = 0; p.ast = 0; p.stl = 0; p.onCourt = false;

  });

  updateDisplay();

}


function saveGame() {

  const game = {

    date: new Date().toLocaleString('es-ES'),

    team1: `${STATE.game.team1.name} ${STATE.game.team1.score}`,

    team2: `${STATE.game.team2.name} ${STATE.game.team2.score}`,

    players: STATE.players.map(p => ({name: p.name, num: p.num, pts: p.pts}))

  };

  STATE.history.unshift(game);

  localStorage.setItem('hoopstat_history', JSON.stringify(STATE.history.slice(0, 50)));

  loadHistory();

}


function loadHistory() {

  const list = document.getElementById('recent-games');

  const histList = document.getElementById('history-list');

  

  if (STATE.history.length === 0) {

    list.innerHTML = '<div style="text-align:center; opacity:0.5;">Sin partidos guardados</div>';

    histList.innerHTML = 'Sin partidos guardados';

    return;

  }

  

  let html = '<h3>Últimos partidos</h3>';

  STATE.history.slice(0, 5).forEach(game => {

    html += `<div class="history-item" onclick="loadGame('${game.date}')">

      <div style="font-weight:700;">${game.date}</div>

      <div style="font-size:20px;">${game.team1} vs ${game.team2}</div>

    </div>`;

  });

  list.innerHTML = html;

  

  let histHtml = '';

  STATE.history.slice(0, 10).forEach(game => {

    histHtml += `<div class="history-item">

      <div style="font-weight:700;">${game.date}</div>

      <div style="font-size:20px;">${game.team1} vs ${game.team2}</div>

    </div>`;

  });

  histList.innerHTML = histHtml;

}


function loadGame(date) {

  const game = STATE.history.find(g => g.date === date);

  if (game) {

    // Cargar datos del partido

    console.log('Cargando:', game);

  }

}


// VÍDEO FUNCTIONS

function loadVideo(event) {

  const file = event.target.files[0];

  const video = document.getElementById('video-player');

  const url = URL.createObjectURL(file);

  video.src = url;

  video.play();

}


function actionVideo(type) {

  console.log('Acción vídeo:', type);

  // Registrar acción con timestamp del vídeo

  STATE.videoTime = document.getElementById('video-player').currentTime;

  // Aquí iría la lógica de stats por vídeo

}


function getAIInsight() {

  document.getElementById('ai-response').style.display = 'block';

  document.getElementById('ai-response').textContent = 

    '🔍 Análisis AI:\n\n' +

    `• Mejor jugador: #4 García (${STATE.players.find(p=>p.num===4)?.pts || 0} pts)\n` +

    `• Líder rebotes: ${STATE.players.reduce((max,p)=>p.reb>max.reb?p:p,STATE.players[0]).name}\n` +

    `• Sugerencia: Rotar quinteto - mantener a ${STATE.players[0].name} en pista`;

}
